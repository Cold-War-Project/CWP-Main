toggle_unsc_sgui_list = {
	scope = country
	ai_is_valid = {
		always = no
	}
	effect = {
		clear_global_variable_list = unsc_members_list
		every_country = {
			limit = {
				has_variable = unsc_member
			}
			add_to_global_variable_list = {
				name = unsc_members_list
				target = THIS
			}
		}
	}
}
