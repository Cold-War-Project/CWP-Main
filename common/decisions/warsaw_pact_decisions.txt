form_warsaw_pact = {
	is_shown = {
		exists = c:RUS
		this = c:RUS
		this = {
			not = {
				has_global_variable = warsaw_pact_leader
			}
		}
	}
	possible = {
		country_rank >= rank_value:great_power
		has_technology_researched = blocs
	}
	when_taken = {
		trigger_event = warsaw_pact_formation.1
		set_global_variable = {
			name = warsaw_pact_leader
			value = ROOT
		}
	}
	ai_chance = {
		base = 100
	}
}

join_warsaw_pact = {
	is_shown = {
		this = {
			is_subject = no
			has_global_variable = warsaw_pact
			country_is_in_europe = yes
			nor = {
				is_target_in_global_variable_list = {
					name = warsaw_pact_members
					target = THIS
				}
				is_target_in_global_variable_list = {
					name = warsaw_pact_application_list
					target = THIS
				}
				has_variable = warsaw_pact_application_rejected
			}
			has_diplomatic_pact = {
				who = c:RUS
				type = diplomatic_alignment
			}
		}
	}
	possible = {
		this = {
			nor = {
				has_diplomatic_pact = {
					who = c:USA
					type = diplomatic_alignment
				}
				is_at_war = yes
			}
			has_diplomatic_pact = {
				who = c:RUS
				type = diplomatic_alignment
			}
		}
	}
	when_taken = {
		this = {
			set_variable = {
				name = application_progress_warsaw_pact
				value = 0
			}
			set_variable = {
				name = warsaw_pact_supporting_list_size
				value = 0
			}
			add_to_global_variable_list = {
				name = warsaw_pact_application_list
				target = THIS
			}
		}
	}
	ai_chance = {
		# TODO, does not work
		base = 10
	}
}

leave_warsaw_pact = {
	is_shown = {
		this = {
			is_target_in_global_variable_list = {
				name = warsaw_pact_members
				target = THIS
			}
		}
	}
	possible = {
		this = {
			c:RUS = {
				relations:root <= relations_threshold:poor
			}
			not = {
				has_diplomatic_pact = {
					who = c:RUS
					type = diplomatic_alignment
				}
			}
		}
	}
	when_taken = {
		this = {
			effect_warsaw_pact_leave = yes
		}
	}
	ai_chance = {
		# TODO, does not work
		base = -30
	}
}
