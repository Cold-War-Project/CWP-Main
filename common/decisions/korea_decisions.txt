join_korean_conflict_ai = {
	is_shown = {
		c:PRC ?= this
		is_player = no
		exists = c:NRK
		exists = c:STK
		NOT = {
			has_war_with = c:STK
			is_in_war_together = c:NRK
		}
		NOT = {
			has_variable = joined_korean_conflict_china
		}
	}
	possible = {
		c:NRK = {
			has_war_with = c:STK
		}
		c:STK.relations:root < relations_threshold:amicable
	}
	when_taken = {
		set_variable = joined_korean_conflict_china
		every_diplomatic_play = {
			limit = {
				OR = {
					AND = {
						initiator = c:NRK
						target = c:STK
					}
					AND = {
						initiator = c:STK
						target = c:NRK
					}
				}
				is_war = yes
			}
			add_initiator_backers = {
				root
			}
		}
	}
	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				any_diplomatic_play = {
					OR = {
						AND = {
							initiator = c:NRK
							target = c:STK
						}
						AND = {
							initiator = c:STK
							target = c:NRK
						}
					}
					is_war = yes
					war = {
						has_war_support = {
							target = c:NRK
							value < 0
						}
					}
				}
			}
			add = 100
		}
	}
}
