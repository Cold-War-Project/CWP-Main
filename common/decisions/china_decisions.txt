propose_armistice_for_chinese_civil_war = {
	is_shown = {
		or = {
			and = {
				exists = c:PRC
				this = c:PRC
			}
			and = {
				exists = C:ROC
				this = c:ROC
			}
		}
		NOT = {
			has_global_variable = china_armistice_proposed
		}
	}
	possible = {
		or = {
			and = {
				exists = c:PRC
				this = c:PRC
			}
			and = {
				exists = c:ROC
				this = c:ROC
			}
		}
		not = {
			has_global_variable = china_armistice_proposed
		}
	}
	when_taken = {
		set_global_variable = china_armistice_proposed
		if = {
			limit = {
				exists = c:PRC
				this = c:PRC
				exists = c:ROC
			}
			c:ROC = {
				trigger_event = {
					id = china_frontline_saving_events.5
					days = 0
					popup = yes
				}
			}
		}
		if = {
			limit = {
				exists = c:PRC
				exists = c:ROC
				this = c:ROC
			}
			c:PRC = {
				trigger_event = {
					id = china_frontline_saving_events.4
					days = 0
					popup = yes
				}
			}
		}
	}
	ai_chance = {
		base = -100
		modifier = {
			trigger = {
				game_date > 1952.1.1
			}
			add = 200
		}
	}
}
