anti_corruption_drive = {
	is_shown = {
		this = {
			has_variable = var_corruption
		}
	}

	possible = {
		custom_tooltip = {
			text = anti_corruption_drive_tooltip
			this = {
				corruption_value >= 40
			}
		}
		not = {
			has_modifier = anti_corruption_drive_conducted
		}

	}

	when_taken = {
		add_modifier = {
			name = anti_corruption_drive_conducted
			months = 60
		}
	}

	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				corruption_value >= 55
			}
			add = 20
		}
		modifier = {
			trigger = {
				corruption_value >= 60
			}
			add = 30
		}
		modifier = {
			trigger = {
				corruption_value >= 70
			}
			add = 40
		}
		modifier = {
			trigger = {
				bureaucracy <= 500
			}
			add = 50
		}
		modifier = {
			trigger = {
				corruption_value >= 80
			}
			add = 1000
		}
		modifier = {
			trigger = { gold_reserves < 0 }
			add = 100
		}
	}
}

anti_corruption_purge = {
	is_shown = {
		this = {
			has_variable = var_corruption
			or = {
				has_law = law_type:law_no_due_process
				has_law = law_type:law_kangaroo_courts
				has_law = law_type:law_politicized_courts
			}
		}
	}

	possible = {
		custom_tooltip = {
			text = anti_corruption_purge_tooltip
			this = {
				corruption_value >= 50
			}
		}
		not = {
			has_modifier = anti_corruption_purge
		}
		or = {
			has_law = law_type:law_no_due_process
			has_law = law_type:law_kangaroo_courts
			has_law = law_type:law_politicized_courts
		}

	}

	when_taken = {
		trigger_event = corruption_events.1
	}

	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				corruption_value >= 55
			}
			add = 10
		}
		modifier = {
			trigger = {
				corruption_value >= 60
			}
			add = 20
		}
		modifier = {
			trigger = {
				corruption_value >= 70
			}
			add = 30
		}
		modifier = {
			trigger = {
				bureaucracy <= 500
			}
			add = 50
		}
		modifier = {
			trigger = {
				corruption_value >= 80
			}
			add = 1000
		}
	}
}
