join_indonesian_federation = {
	is_shown = {
		exists = c:IDS
		NOT = {
			this = c:IDS
		}
		this = {
			capital.region = sr:region_indonesia
			any_primary_culture = {
				is_primary_culture_of = c:IDS
			}
		}
	}
	possible = {
		OR = {
			is_subject = no
			is_subject_of = c:IDS
		}
		NOT = {
			has_truce_with = c:IDS
		}
		has_strategic_adjacency = c:IDS
	}
	when_taken = {
		if = {
			limit = {
				is_player = yes
			}
			play_as = c:IDS
		}
		c:IDS = {
			annex = root
		}
	}
	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				is_subject_of = c:IDS
			}
			add = 50
		}
		modifier = {
			trigger = {
				religion = c:IDS.religion
			}
			add = 50
		}
		modifier = {
			trigger = {
				is_adjacent = c:IDS
			}
			add = 25
		}
		modifier = {
			trigger = {
				c:IDS.relations:root > 50
			}
			add = 25
		}
		modifier = {
			trigger = {
				c:IDS.relations:root > 80
			}
			add = 25
		}
		modifier = {
			trigger = {
				c:IDS.relations:root < 0
			}
			add = -25
		}
	}
}

merge_austrias = {
	is_shown = {
		exists = c:AUS
		exists = c:AOA
		OR = {
			this = c:AUS
			this = c:AOA
		}
		OR = {
			AND = {
				c:AOA = {
					is_subject = no
				}
				c:AUS = {
					is_subject = no
				}
			}
			any_country = {
				c:AUS = {
					is_subject_of = PREV
				}
				c:AOA = {
					is_subject_of = PREV
				}
			}
		}
	}
	possible = {
		trigger_if = {
			limit = {
				this = c:AUS
			}
			c:AUS.num_states < c:AOA.num_states
		}
		trigger_else_if = {
			limit = {
				this = c:AOA
			}
			c:AOA.num_states < c:AUS.num_states
		}
		trigger_else = {
			always = yes
		}
	}
	when_taken = {
		if = {
			limit = {
				this = c:AUS
			}
			if = {
				limit = {
					is_player = yes
				}
				play_as = c:AOA
			}
			c:AOA = {
				annex = c:AUS
			}
		}
		if = {
			limit = {
				this = c:AOA
			}
			if = {
				limit = {
					is_player = yes
				}
				play_as = c:AUS
			}
			c:AUS = {
				annex = c:AOA
			}
		}
	}
	ai_chance = {
		base = 100
	}
}
