form_nato = {
	is_shown = {
		exists = c:USA
		this = c:USA
		this = {
			not = {
				has_global_variable = nato_leader
			}
		}
	}
	possible = {
		country_rank >= rank_value:superpower
		has_technology_researched = blocs
	}
	when_taken = {
		trigger_event = nato_formation.1
		set_global_variable = {
			name = nato_leader
			value = ROOT
		}
	}
	ai_chance = {
		base = 100
	}
}

join_nato = {
	is_shown = {
		exists = c:USA
		this = {
			is_subject = no
			has_global_variable = nato
			or = {
				country_is_in_europe = yes
				c:CAN ?= this

			}
			nor = {
				is_target_in_global_variable_list = {
					name = nato_members
					target = THIS
				}
				is_target_in_global_variable_list = {
					name = nato_application_list
					target = THIS
				}
				has_variable = nato_application_rejected
			}
			has_diplomatic_pact = {
				who = c:USA
				type = diplomatic_alignment
			}
		}
	}
	possible = {
		this = {
			nor = {
				AND = {
					exists = c:RUS
					has_diplomatic_pact = {
						who = c:RUS
						type = diplomatic_alignment
					}
				}
				is_at_war = yes
			}
			has_diplomatic_pact = {
				who = c:USA
				type = diplomatic_alignment
			}
		}
	}
	when_taken = {
		this = {
			set_variable = {
				name = application_progress_nato
				value = 0
			}
			set_variable = {
				name = nato_supporting_list_size
				value = 0
			}
			add_to_global_variable_list = {
				name = nato_application_list
				target = THIS
			}
		}
	}
	ai_chance = {
		# TODO, does not work
		base = 10
	}
}

leave_nato = {
	is_shown = {
		this = {
			is_target_in_global_variable_list = {
				name = nato_members
				target = THIS
			}
		}
	}
	possible = {
		this = {
			exists = c:USA
			c:USA = {
				relations:root <= relations_threshold:poor
			}
			not = {
				has_diplomatic_pact = {
					who = c:USA
					type = diplomatic_alignment
				}
			}
		}
	}
	when_taken = {
		this = {
			effect_nato_leave = yes
		}
	}
	ai_chance = {
		# TODO, does not work
		base = -30
	}
}
