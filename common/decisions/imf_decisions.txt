join_imf = {
	is_shown = {
		this = {
			not = {
				is_target_in_global_variable_list = {
					name = imf_members
					target = THIS
				}
			}
		}
	}
	possible = {
		has_technology_researched = international_monetary_system
		is_at_war = no
		gold_reserves >= 15000
	}
	when_taken = {
		add_treasury = -15000
		custom_tooltip = {
			text = WILL_BECOME_IMF_MEMBER
			set_imf_member = yes
			set_imf_starting_values = {
				debtValue = 0
				creditValue = 0
				contributionValue = 15000
				debtYearsValue = 0
				interestRateValue = 2.5
				loanType = 1
			}
			set_variable = {
				name = joined_imf_recently
				years = 5
			}
		}
	}
	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				country_is_socialist = yes
			}
			add = -30
		}
		modifier = {
			trigger = {
				country_is_isolationist = yes
			}
			add = -30
		}
		modifier = {
			trigger = {
				country_is_capitalist = yes
			}
			add = 30
		}
	}
}

leave_imf = {
	is_shown = {
		this = {
			is_target_in_global_variable_list = {
				name = imf_members
				target = THIS
			}
		}
	}
	possible = {
		custom_tooltip = {
			text = HAS_NO_IMF_DEBT
			var:imf_debt <= 0
		}
		custom_tooltip = {
			text = HAS_NO_IMF_CREDIT
			var:imf_credit <= 0
		}
		custom_tooltip = {
			text = IMF_CONTRIBUTION_SHARE_LESS_THAN
			imf_contribution_share <= 0.1
		}
		not = {
			custom_tooltip = {
				text = JOINED_IMF_RECENTLY
				has_variable = joined_imf_recently
			}
		}
	}
	when_taken = {
		custom_tooltip = {
			text = WILL_LEAVE_IMF
			remove_imf_member = yes
		}
	}
	ai_chance = {
		base = -100
		modifier = {
			trigger = {
				country_is_isolationist = yes
			}
			add = 100
		}
	}
}
