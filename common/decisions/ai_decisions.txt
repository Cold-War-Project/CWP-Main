create_olympic_comittee_ai = {
	is_shown = {
		this = {
			is_ai = yes
			nor = {
				is_target_in_global_variable_list = { name = national_olympic_committee_list target = THIS}
				is_target_in_global_variable_list = { name = unrecognized_olympic_committee_list target = THIS}
			}
		}
	}

	possible = {
		this = {
			is_ai = yes
			is_at_war = no
			nor = {
				is_target_in_global_variable_list = { name = national_olympic_committee_list target = THIS}
				is_target_in_global_variable_list = { name = unrecognized_olympic_committee_list target = THIS}
			}
		}
	}

	when_taken = {
		this = {
   			if = {
     			limit = {
					overlord = { is_target_in_global_variable_list = { name = un_members_list target = THIS} }
   		    	}
      	  		add_to_global_variable_list = {
      	      		name = national_olympic_committee_list
       	     		target = THIS
       	 		}
   	 		}
   			else = {
    	    	add_to_global_variable_list = {
    	       		name = unrecognized_olympic_committee_list
     	       		target = THIS
     	   		}
  		  	}
            set_variable = application_progress_olympics
		}
	}

	ai_chance = {
		base = 1
		modifier = { 
			trigger = {
				is_target_in_global_variable_list = { name = un_members_list target = THIS}
			}
			add = 1000
		}
	}
}
