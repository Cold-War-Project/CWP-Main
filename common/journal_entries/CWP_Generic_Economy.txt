# Housing For The People
je_housing = {
	icon = "gfx/interface/icons/event_icons/event_trade.dds"
	is_shown_when_inactive = {
		is_player = yes
		NOT = {
			has_variable = housing_je_done
			pop_type_percent_country = {
				pop_type = informal_workers
				percent > 0.6
			}
		}
		any_scope_building = {
			is_building_type = building_urban_center
			level >= 5
		}
	}
	immediate = {
		if = {
			limit = {
				market_capital.market = {
					mg:housing = {
						market_goods_cheaper < 0
					}
				}
			}
			set_variable = {
				name = market_goods_more_expensive
			}
		}
		else_if = {
			limit = {
				market_capital.market = {
					mg:housing = {
						market_goods_cheaper > 0
					}
				}
			}
			set_variable = {
				name = market_goods_cheaper
			}
		}
	}
	possible = {
		any_scope_building = {
			is_building_type = building_housing
		}
	}
	complete = {
		trigger_if = {
			any_scope_state = {
				percent >= 0.95
				has_building = building_housing
				any_scope_building = {
					is_building_type = building_housing
					occupancy < 0.95
					level >= 10
				}
			}
			any_scope_state = {
				percent >= 0.9
				OR = {
					NOT = {
						has_building = building_slums
					}
					any_scope_building = {
						is_building_type = building_slums
						occupancy <= 0.1
					}
				}
			}
			market_capital.market = {
				mg:housing = {
					market_goods_cheaper > 0
				}
			}
		}
	}
	on_complete = {
		trigger_event = {
			id = cwp_generic_economy_events.1
		}
		set_variable = {
			name = housing_je_done
		}
	}
}
