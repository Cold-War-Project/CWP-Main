je_imf_concessionary_loan = {
	icon = "gfx/interface/icons/cwp_icons/imf_seal.dds"
	
	group = je_group_foreign_affairs
	
	should_be_pinned_by_default = yes
	possible = {
		custom_tooltip = {
			text = is_imf_member
			is_target_in_global_variable_list = {
				name = imf_members
				target = THIS
			}
		}
		imf_debt_value > 0
		nor = {
			has_journal_entry = je_imf_austerity_measures
			has_journal_entry = je_imf_bailout_loan
		}
	}
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = not_paying_imf_loans_status
				trigger = {
					var:imf_debt_monthly_percentage <= 0 
				}
			}
			triggered_desc = {
				desc = paying_imf_loans_status
				trigger = {
					var:imf_debt_monthly_percentage >= 0.01 
				}
			}
		}
	}
	immediate = {
		set_variable = {
			name = imf_loan_payment_months_current
			value = 0
		}
	}
	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = imf_loan_payment_months_current
				add = 1
			}
		}
	}
	current_value = {
		value = var:imf_loan_payment_months_current
	}
	goal_add_value = {
		add = imf_application_payment_due_months
	}
	on_timeout = {
	
	}
	complete = {
		custom_tooltip = {
			text = IMF_LOAN_PAID
			imf_debt_value <= 0 
		}
	}
	fail = {
		or = {
			scope:journal_entry = {
				is_goal_complete = yes
			}
			has_modifier = defaulted_on_international_loan
		}
	}
	on_complete = {
		custom_tooltip = {
			text = IMF_LOAN_PAID_CONCLUSION
			imf_loan_conclusion = yes
		}

	}
	on_fail = {
		trigger_event = {
			id = imf_events.3
		}
	}
	invalid = {
		has_modifier = imf_austerity_measures
	}
	progressbar = yes
	weight = 100
}

je_imf_bailout_loan = {
	icon = "gfx/interface/icons/cwp_icons/imf_seal.dds"
	
	group = je_group_foreign_affairs
	
	should_be_pinned_by_default = yes
	possible = {
		and = {
			custom_tooltip = {
				text = is_imf_member
				is_target_in_global_variable_list = {
					name = imf_members
					target = THIS
				}
			}
			has_modifier = imf_bailout
		}

	}
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = has_not_met_imf_standards
				trigger = {
					has_met_imf_requirements = no
				}
			}
		}
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					not = {
						has_modifier = imf_bailout
					}
				}
				add_modifier = {
					name = imf_bailout
				}
			}
		}
	}
	complete = {
		has_met_imf_requirements = yes
	}
	on_complete = {
		custom_tooltip = {
			text = IMF_LOAN_PAID_CONCLUSION
			imf_loan_conclusion = yes
		}
	}
	weight = 100
}
