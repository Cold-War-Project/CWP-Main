je_great_retreat = {
	icon = "gfx/interface/icons/lens_toolbar_icons/nuclear_weapons.dds"
	should_be_pinned_by_default = yes
	possible = {
		and = {
			exists = c:ROC
			this = c:ROC
		}
		c:PRC = {
			has_war_with = ROOT
		}
	}
	immediate = {
		set_variable = {
			name = retreat_level
			value = 0
		}
		this = {
			set_variable = great_retreat
		}
		c:PRC = {
			save_scope_as = PRC_SCOPE
		}
		c:ROC = {
			save_scope_as = ROC_SCOPE
		}
	}
	current_value = {
		value = var:retreat_level
	}
	goal_add_value = {
		add = 48
	}
	on_weekly_pulse = {
		effect = {
			change_variable = {
				name = retreat_level
				add = 1
			}
		}
	}
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}
	fail = {
		not = {
			or = {
				s:STATE_FORMOSA.region_state:ROC = {
					controller = c:ROC
				}
				c:ROC = {
					owns_entire_state_region = STATE_FORMOSA
				}
			}
		}
	}
	on_complete = {
		remove_variable = retreat_level
		custom_tooltip = {
			text = "WILL_GIVE_UP_MAIN_LAND_ROC"
			effect_great_retreat_roc = yes
			effect_free_all_subjects = yes
		}
		s:STATE_FORMOSA.region_state:ROC = {
			remove_modifier = kuomintang_relocation
		}
		every_diplomatic_play = {
			limit = {
				initiator_is = c:PRC
				target_is = c:ROC
			}
			end_play = yes
		}
	}
	on_fail = {
		remove_variable = retreat_level
		s:STATE_FORMOSA.region_state:ROC = {
			remove_modifier = kuomintang_relocation
		}
	}
	can_deactivate = yes
	progressbar = yes
	weight = 100
}
