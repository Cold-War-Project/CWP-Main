administer = {
	requires_approval = no
	should_notify_third_parties = no
	unlocking_technologies = {
		international_relations
	}
	potential = {
		NOT = {
			is_country_type = decentralized
		}
		scope:target_country = {
			NOT = {
				is_country_type = decentralized
			}
			is_subject_of = root
		}
	}
	possible = {
		NOT = {
			has_war_with = scope:target_country
		}
		has_diplomatic_relevance = scope:target_country
		is_revolutionary = no
		in_default = no
	}
	pact = {
		cost = 10
		is_two_sided_pact = no
		show_in_outliner = yes
		relations_progress_per_day = 1		# At 100, +1 relations
		relations_improvement_max = 80
		weekly_effect = {
			custom_tooltip = {
				text = administer_weekly_effect_tooltip
					set_local_variable = {
						name = colonial_administration_amount
						value = {
							value = scope:target_country.total_population
							divide = 100000
							round = yes
						}
					}
					if = {
						limit = {
							has_local_variable = colonial_administration_amount
						}
						root = {
							add_modifier = {
								name = colonial_administration
								days = 7
								multiplier = local_var:colonial_administration_amount
							}
						}
						scope:target_country = {
							add_modifier = {
								name = colonial_administrators
								days = 7
								multiplier = local_var:colonial_administration_amount
							}
						}
					}
				}
			}
		is_about_to_auto_break = {
		}
		should_auto_break = {
		}
		should_invalidate = {
			OR = {
				has_war_with = scope:target_country
				is_country_type = decentralized
				scope:target_country = {
					OR = {
						is_country_type = decentralized
						NOT = {
							is_subject_of = root
						}
					}
				}
				in_default = yes
			}
		}
	}
	ai = {
		max_influence_spending_fraction = 0.2
		propose_score = {
			value = 10
		}
		will_propose = {
			scaled_debt <= 0
			is_at_war = no
			has_attitude = {
				who = scope:target_country
				attitude = protective
			}
		}
		will_break = {
			OR = {
				scaled_debt > 0.25
				AND = {
					scaled_debt > 0
					is_at_war = yes
				}
				NOT = {
					has_attitude = {
						who = scope:target_country
						attitude = protective
					}
				}
				
			}
		}
	}
}
