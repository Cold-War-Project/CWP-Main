		}

		add = {
			desc = "POP_CLERGY"
			if = {
				limit = {
					is_pop_type = clergy

			}
		}

			}
		}

			desc = "LEADER_POPULARITY"
			scope:interest_group = {

					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
