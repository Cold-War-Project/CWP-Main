        }
        value = var:aroai_building_university_spending
        floor = yes
        divide = {
            value = var:aroai_building_university_spending
            subtract = {
                value = var:aroai_building_university_spending

            }
            multiply = 100
        }
    }
    else = {
        value = aroai_country_active_income

        multiply = 1.10

    min = 0
    max = 1000000000 #billion
}
