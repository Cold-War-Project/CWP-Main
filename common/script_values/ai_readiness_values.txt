readiness = {
	if = {
		limit = {
			exists = var:var_readiness
		}
		value = var:var_readiness
		min = 0
		max = 100
	}
	else = {
		value = 0
	}
}

readiness_increase_value = {
	if = {
		limit = {
			has_variable = var_increase_readiness
		}
		add = 2.5
	}
}

readiness_decrease_value = {
	if = {
		limit = {
			has_variable = var_decrease_readiness
		}
		add = 2.5
	}
}

readiness_sum_value = {
	if = {
		limit = {
			readiness < 100
		}
		add = {
			value = readiness_increase_value
			desc = ""
			format = ""
		}
	}
	if = {
		limit = {
			readiness > 0
		}
		subtract = {
			value = readiness_decrease_value
			desc = ""
			format = ""
		}
	}
}
ai_increase_readiness = {
		# Increases Readiness
		if = {
			limit = {
				is_at_war = yes
			}
			add = 50
		}
		if = {
			limit = {
				is_diplomatic_play_target = yes
			}
			add = 20
		}
		if = {
			limit = {
				is_diplomatic_play_initiator = yes
			}
			add = 30
		}
}

ai_decrease_readiness = {
	# Increases Readiness
	value = 10
	if = {
		limit = {
			is_at_war = no
		}
		add = 50
	}
	if = {
		limit = {
			is_diplomatic_play_target = no
		}
		add = 20
	}
}

ai_ideal_readiness = {
	# Increases Readiness
	add = ai_increase_readiness
	# Decreases Readiness
	subtract = ai_decrease_readiness
	min = ai_ideal_readiness_min
	max = 100
}

ai_ideal_readiness_min = {
	if = {
		limit = { # Superpowers maintain more readiness
			country_rank = rank_value:superpower
		}
		add = 20
	}
	if = {
		limit = { # Great Powers maintain more readiness
			country_rank = rank_value:great_power
		}
		add = 10
	}
	if = {
		limit = {
			or = {
				is_target_in_global_variable_list = {
					name = warsaw_pact_application_list
					target = THIS
				}
				is_in_warsaw_pact = yes
			}
		}
		add = warsaw_pact_mandated_readiness
	}
	if = {
		limit = {
			or = {
				is_target_in_global_variable_list = {
					name = nato_application_list
					target = THIS
				}
				is_in_nato = yes
			}
		}
		add = nato_mandated_readiness
	}
}
