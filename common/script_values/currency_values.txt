currency_value_growth = {
	value = 0
	if = {
		limit = {
			or = {
				money_supply > 5 
				money_supply < 5 
			}
			nor = {
				has_variable = expansionary_monetary_policy
				has_variable = contractionary_monetary_policy
			}
		}
		subtract = {
			value = money_supply_growth_percentage
			divide = 10
			desc = CUR_MONEY_SUPPLY
			format = "CUR_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = {
			has_variable = expansionary_monetary_policy
		}
		subtract = {
			value = money_supply_growth_percentage
			divide = 10
			desc = MNS_EXPANSIONARY
			format = "CUR_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = {
			has_variable = contractionary_monetary_policy
		}
		subtract = {
			value = money_supply_growth_percentage
			divide = 10
			desc = MNS_CONTRACTIONARY
			format = "CUR_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = {
			interest_rate >= 1
			not = {
				has_variable = pegged_currency
			}
		}
		add = {
			value = interest_rate
			divide = 40
			desc = CUR_INTEREST_RATE
			format = "CUR_TOOLTIP_FORMAT"
		}
	}
	if = {
		# If interest is lower than 1 we want to simulate that the interest rates stops working 
		limit = {
			interest_rate < 1
			not = {
				has_variable = pegged_currency
			}
		}
		subtract = {
			value = interest_rate
			divide = 1600
			desc = CUR_INTEREST_RATE
			format = "CUR_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = {
			not = {
				has_variable = pegged_currency
			}
			currency_average_difference_to_global_AI > currency_value_max
		}
		subtract = {
			value = currency_value_ratio
			divide = 2
			desc = CUR_SIZE_OF_ECONOMY
			format = "CUR_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = {
			has_variable = pegged_currency
		}
		value = {
			value = var:pegged_currency.currency_value_growth
			desc = CUR_PEGGED_CURRENCY
			format = "CUR_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = {
			is_leading_central_bank = no
		}
		value = 0
	}
}
