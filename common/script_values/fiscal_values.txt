bank_credibility_val = {
	value = 0
	# From Altering Currency
	if = {
		limit = {
			has_variable = altered_currency
		}
		subtract = {
			value = bank_credibility_10
			multiply = 3
			desc = BNK_ALTERED_CURRENCY
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	# From Gold Reserves
	if = {
		limit = {
			gold_reserve_ratio >= 0.6
		}
		subtract = {
			value = gold_reserve_ratio
			multiply = 3
			desc = BNK_GOLD_RESERVES
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	# From debt
	if = {
		limit = {
			scaled_debt >= 0.6
		}
		subtract = {
			value = scaled_debt
			multiply = 5
			desc = BNK_DEBT
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	# From Income
	if = {
		limit = {
			income >= 1
		}
		add = {
			value = income_to_investor
			desc = BNK_INCOME
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	# From keeping stable inflation
	if = {
		limit = { 
			inflation_growth <= 3
			inflation_growth >= -3
		}
		add = {
			value = bank_credibility_10
			multiply = 2
			desc = BNK_INFLATION_GROWTH
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = { 
			inflation_growth >= 3
			inflation_growth <= -3
		}
		subtract = {
			value = bank_credibility_10
			multiply = 2
			desc = BNK_INFLATION_GROWTH
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = { 
			inflation_value <= 10 
			inflation_value >= 0 
		}
		add = {
			value = bank_credibility_10
			multiply = 5
			desc = BNK_INFLATION_LEVEL
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	# From Inflation
	if = {
		limit = { 
			inflation_value > 10
		}
		subtract = {
			value = inflation_value
			divide = 5
			round = yes
			desc = BNK_INFLATION_LEVEL
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	# From Deflation
	if = {
		limit = { inflation_value < 0 }
		subtract = {
			value = bank_credibility_10
			multiply = 5
			desc = BNK_DEFLATION_LEVEL
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	# From Bank Independence
	if = {
		limit = { has_law = law_type:law_government_controlled }
		subtract = {
			value = bank_credibility_10
			multiply = 3
			desc = BNK_GOVERNMENT_CONTROLLED
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = { has_law = law_type:law_government_interference }
		subtract = {
			value = bank_credibility_10
			multiply = 2
			desc = BNK_GOVERNMENT_INTERFERENCE
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = { has_law = law_type:law_independent_monetary_policy}
		add = {
			value = bank_credibility_10
			multiply = 3
			desc = BNK_INDEPENDENT_MONETARY_POLICY
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	min = 5
	# From Debt	
	if = {
		limit = {
			scaled_debt >= 0.8
			income <= 1
		}
		value = {
			value = 4
			desc = BNK_IMMINENT_DEFAULT
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = {
			scaled_debt >= 0.94
			income <= 1
		}
		value = {
			value = 3
			desc = BNK_IN_DEFAULT
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = {
			scaled_debt >= 0.96
			income <= 1
		}
		value = {
			value = 2
			desc = BNK_IN_DEFAULT
			format = "BNK_TOOLTIP_FORMAT"
		}
	}
	if = {
		limit = {
			scaled_debt >= 0.98
			income <= 1
		}
		value = 1
			desc = BNK_IN_DEFAULT
			format = "BNK_TOOLTIP_FORMAT"
	}
	round = yes
	max = 24
}

	value = {

	}

			has_law = law_type:law_government_controlled
		}

	}
	if = {
		limit = {
			has_law = law_type:law_government_interference
		}
		multiply = 0.6
	}
	if = {
		limit = {
			has_law = law_type:law_independent_monetary_policy
		}
		multiply = 1.5
	}
	### This represents the 'zero lower bound', where interest rates lose effectiveness rapidly below zero percent
	if = {
		limit = {
			interest_rate < 0
		}
		multiply = {
			value = 1
			add = {
				value = interest_rate
				multiply = 0.01
			}
			min = 0.1
		}
	}
}
