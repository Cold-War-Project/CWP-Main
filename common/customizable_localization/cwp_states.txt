	type = state
	text = {
		trigger = {
			owner = {
				country_definition = {
					any_primary_culture = {

					}
				}
				NOT = {
					this = c:POL
				}
			}
		}
		localization_key = dyn_state_kralovec_rus
	}
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:polish
			}
		}
		localization_key = dyn_state_kralovec_pol
	}
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:north_german
			}
		}
		localization_key = dyn_state_kralovec_ger
	}
	text = {
		trigger = {

		}
		localization_key = dyn_state_kralovec
	}
