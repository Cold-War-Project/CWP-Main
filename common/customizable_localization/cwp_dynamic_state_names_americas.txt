### Baja California

dyn_state_baja_california = {
	type = state
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:mexican
			}
		}
		localization_key = dyn_state_baja_california_mex
	}
	text = {
		trigger = {
			owner = {
				country_definition = {
					any_primary_culture = {
						shares_non_heritage_trait_with_any_primary_culture = c:RUS
					}
				}
			}
		}
		localization_key = dyn_state_baja_california_rus
	}
	text = {
		localization_key = dyn_state_baja_california
		fallback = yes
	}
}

### California

dyn_state_california = {
	type = state
	text = {
		trigger = {
			owner = {
				and = {
					country_definition = {
						any_primary_culture = {
							shares_non_heritage_trait_with_any_primary_culture = c:RUS
						}
					}
					NOT = {
						this = c:POL
					}
				}
			}
		}
		localization_key = dyn_state_california_rus
	}
	text = {
		trigger = {
			owner = {
				OR = {
					country_definition = {
						any_primary_culture = {
							shares_non_heritage_trait_with_any_primary_culture = c:PRC
						}
					}
					country_definition = {
						any_primary_culture = {
							shares_non_heritage_trait_with_any_primary_culture = c:ROC
						}
					}
				}
			}
		}
		localization_key = dyn_state_california_han
	}
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:french
			}
		}
		localization_key = dyn_state_california_fra
	}
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:japanese
			}
		}
		localization_key = dyn_state_california_jap
	}
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:korean
			}
		}
		localization_key = dyn_state_california_kor
	}
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:polish
			}
		}
		localization_key = dyn_state_california_pol
	}
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:portuguese
			}
		}
		localization_key = dyn_state_california_por
	}
	text = {
		localization_key = dyn_state_california
		fallback = yes
	}
}

### Washington

dyn_state_washington = {
	type = state
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:russian
			}
		}
		localization_key = dyn_state_washington_rus
	}
	text = {
		trigger = {
			owner = {
				this = c:PRC
			}
		}
		localization_key = dyn_state_washington_prc
	}
	text = {
		trigger = {
			owner = {
				and = {
					OR = {
						country_definition = {
							any_primary_culture = {
								shares_non_heritage_trait_with_any_primary_culture = c:ROC
							}
						}
						country_has_primary_culture = cu:han
					}
					NOT = {
						this = c:PRC
					}
				}
			}
		}
		localization_key = dyn_state_washington_roc
	}
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:french
			}
		}
		localization_key = dyn_state_washington_fra
	}
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:finnish
			}
		}
		localization_key = dyn_state_washington_fin
	}
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:russian
			}
		}
		localization_key = dyn_state_washington_rus
	}
	text = {
		trigger = {
			owner = {
				OR = {
					country_has_primary_culture = cu:dutch
					country_has_primary_culture = cu:flemish
					country_has_primary_culture = cu:wallonian
				}
				
			}
		}
		localization_key = dyn_state_washington_hol
	}
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:polish
			}
		}
		localization_key = dyn_state_washington_pol
	}
	text = {
		trigger = {
			owner = {
				AND = {
					OR = {
						country_definition = {
							any_primary_culture = {
								shares_non_heritage_trait_with_any_primary_culture = c:SPA
							}
						}
						country_has_primary_culture = cu:spanish
					}
					NOT = {
						this = c:MEX
					}
				}
				
			}
		}
		localization_key = dyn_state_washington_spa
	}
	text = {
		trigger = {
			owner = {
				country_has_primary_culture = cu:swedish
			}
		}
		localization_key = dyn_state_washington_swe
	}
	text = {
		localization_key = dyn_state_washington
		fallback = yes
	}
}
