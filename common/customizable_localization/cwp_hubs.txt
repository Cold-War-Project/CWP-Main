dyn_hub_wood = {
	type = state
	# STATE_KRALOVEC
	text = {
		trigger = {
			state_region = s:STATE_KRALOVEC
			owner = {
				and = {
					country_definition = {
						any_primary_culture = {
							shares_non_heritage_trait_with_any_primary_culture = c:RUS
						}
					}
					NOT = {
						this = c:POL
					}
				}
			}
		}
		localization_key = dyn_hub_wood_kralovec_rus
	}
	text = {
		trigger = {
			state_region = s:STATE_KRALOVEC
			owner = {
				country_has_primary_culture = cu:polish
			}
		}
		localization_key = dyn_hub_wood_kralovec_pol
	}

		trigger = {
			state_region = s:STATE_KRALOVEC

		}
