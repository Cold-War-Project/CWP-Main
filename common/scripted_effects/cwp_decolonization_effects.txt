		limit = {

		}

					add_primary_culture = prev

		limit = {

				limit = {
					exists = scope:overlord_culture

				NOT = {

		}
		root = {
			remove_primary_culture = prev
		}
	}

		set_state_type = unincorporated

				is_subject_type = subject_type_puppet
				is_adjacent_to_country = prev

	}
