activate_un_decolonization_je = {
	every_country = {
		if = {
			limit = {
				or = {
					has_variable = unsc_member
					has_variable = un_member
				}
			}
			add_journal_entry = {
				type = je_decolonization_resolution
			}
		}
	}
	set_global_variable = pending_resolution
	set_global_variable = {
		name = pending_resolution_progress
		value = 0
	}
}
