effect_unsc_member = {
	set_variable = {
		name = unsc_member
		value = 0
	}
}

effect_un_member = {
	set_variable = {
		name = un_member
		value = 0
	}
}

effect_reset_un_lists = {
	set_global_variable = {
		name = is_supporting_list
		value = 0
	}
	set_global_variable = {
		name = is_opposing_list
		value = 0
	}
	set_global_variable = {
		name = pending_resolution_progress
		value = 0
	}
	remove_global_variable = vetoed
	remove_global_variable = pending_resolution
	every_country = {
		if = {
			limit = {
				has_variable = is_supporting
			}
			remove_variable = is_supporting
		}
		if = {
			limit = {
				has_variable = is_opposing
			}
			remove_variable = is_opposing
		}
	}
}
