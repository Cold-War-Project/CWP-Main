effect_un_application_approved = {
	remove_variable = un_observer
	remove_variable = un_supporting_list_size
	remove_variable = application_progress_un
	clear_variable_list = un_supporting
	clear_variable_list = un_abstaining
	clear_variable_list = un_rejecting
	remove_list_global_variable = {
		name = un_application_list
		target = THIS
	}
	set_variable = un_member
	set_country_type = recognized
}
effect_un_application_rejected = {
	remove_variable = un_supporting_list_size
	remove_variable = application_progress_un
	clear_variable_list = un_supporting
	clear_variable_list = un_abstaining
	clear_variable_list = un_rejecting
	remove_list_global_variable = {
		name = un_application_list
		target = THIS
	}
	set_variable = {
		name = un_application_rejected
		days = 365
	}
}

effect_reset_un_lists = {
	set_global_variable = {
		name = pending_resolution_progress
		value = 0
	}
	every_country = {
		if = {
			limit = {
				has_variable = is_supporting
			}
			remove_variable = is_supporting
		}
		if = {
			limit = {
				has_variable = is_opposing
			}
			remove_variable = is_opposing
		}
	}
	remove_global_variable = asking_un_intervention_country
	remove_global_variable = vetoed
	remove_global_variable = pending_resolution
}

##### Rebuild UN list
rebuild_un_memberlist = {
	clear_global_variable_list = un_members_list
	every_country = {
		limit = {
			has_variable = un_member
		}
		add_to_global_variable_list = {
			name = un_members_list
			target = THIS
		}
	}
}
