### To be used in history
effect_corruption_low = {
	trigger_event = {
		id = debug.101
		days = 1
	}
}

effect_corruption_medium = {
	trigger_event = {
		id = debug.102
		days = 1
	}
}

effect_corruption_high = {
	trigger_event = {
		id = debug.103
		days = 1
	}
}

calc_ig_corruption = {
	if = {
		limit = {
			NOT = {
				has_variable = ig_time_in_government
			}
		}
		set_variable = {
			name = ig_time_in_government
			value = 0
		}
	}
	if = {
		limit = {
			is_in_government = yes
		}
		change_variable = {
			name = ig_time_in_government
			add = 1
		}
	}
	else = {
		change_variable = {
			name = ig_time_in_government
			add = -2
		}
	}
	clamp_variable = {
		name = ig_time_in_government
		max = 240
		min = 0
	}
	if = {
		limit = {
			NOT = {
				has_variable = ig_corruption
			}
		}
		set_variable = {
			name = ig_corruption
			value = {
				value = ig_corruption_growth
				multiply = 60				# If we don't have any corruption value yet, we start at current growth * 60
			}
		}
	}
	change_variable = {
		name = ig_corruption
		add = ig_corruption_growth
	}
	clamp_variable = {
		name = ig_corruption
		max = 100
		min = 0
	}
}
