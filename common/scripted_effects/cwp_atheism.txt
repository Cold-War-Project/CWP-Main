effect_atheism_activate = {
	activate_law = law_type:law_state_atheism
	set_state_religion = rel:atheist
	every_scope_pop = {
		limit = {
			is_pop_type = technicians
		}
		change_pop_religion = {
			target = rel:atheist
			value = 0.15
		}
	}
	every_scope_pop = {
		limit = {
			is_pop_type = administrators
		}
		change_pop_religion = {
			target = rel:atheist
			value = 0.25
		}
	}
	every_scope_pop = {
		limit = {
			is_pop_type = academics
		}
		change_pop_religion = {
			target = rel:atheist
			value = 0.35
		}
	}
	every_scope_pop = {
		limit = {
			and = {
				literacy_rate > 0.6
				NOT = {
					is_pop_type = clergy
				}
			}
		}
		change_pop_religion = {
			target = rel:atheist
			value = 0.15
		}
	}
}
