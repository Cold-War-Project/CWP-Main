on_action_currency_denomination_change_chance = {
	if = {
		limit = {
			inflation_value > 50
			country_currency_denomination < 10000000
		}
		random = {
			chance = 15
			modifier = {
				trigger = {
					inflation_value > 55
				}
				add = 5
			}
			modifier = {
				trigger = {
					inflation_value > 60
				}
				add = 5
			}
			modifier = {
				trigger = {
					inflation_value > 75
				}
				add = 5
			}
			modifier = {
				trigger = {
					inflation_value > 80
				}
				add = 5
			}
			modifier = {
				trigger = {
					inflation_value > 90
				}
				add = 5
			}
			effect_redenominate = {
				denominationEffect = change_global_variable
				mathAction = multiply
				mathValue = 10
			}
		}
	}
}
