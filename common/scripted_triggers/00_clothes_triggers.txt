				pop_weight_modifier_scale >= 95
			}
			AND = {
				standard_of_living >= 40
				pop_weight_modifier_scale >= 90
			}
		}
	}
}

	scope:pop = {
		standard_of_living < 10
	}
}

	}
}

	exists = scope:pop
	scope:pop = {
		OR = {
			is_pop_type = farmers

			is_pop_type = laborers
		}
		OR = {
			culture = cu:amazonian
			culture = cu:nilotes
		}
	}
}
