gbr_devalue_currency_button = {
	name = "gbr_devalue_currency_button_name"
	desc = "gbr_devalue_currency_button_desc"
	
	visible = {
		has_variable = gbr_started_rationing
	}
	
	possible = {
		hidden_trigger = {
			inflation_value >= 10
			NOT = { has_variable = gbr_devalue_button_pressed }
		}
	}
	
	effect = {
		hidden_effect = {
			trigger_event = united_kingdom_post_war_events.101
			set_variable = gbr_devalue_button_pressed
		}
		if = {
			limit = {
				country_currency_val > 0
			}
			change_variable = {
				name = money_supply
				subtract = {
					value = money_supply_from_gold_reserves
					multiply = 10
				}
			}
		}
		set_variable = {
			name = altered_currency
			months = 12
		}
	}
}

gbr_end_rationing_early_button = {
	name = "gbr_end_rationing_early_button_name"
	desc = "gbr_end_rationing_early_button_desc"
	
	visible = {
		has_variable = gbr_started_rationing
	}
	
	possible = {
		hidden_trigger = {
			OR = {
				AND = {
					var:gbr_months_of_rationing >= 24
					government_legitimacy >= 80
				}
				AND = {
					var:gbr_months_of_rationing >= 48
					government_legitimacy >= 60
				}
				AND = {
					var:gbr_months_of_rationing >= 72
					government_legitimacy >= 40
				}
				AND = {
					var:gbr_months_of_rationing >= 96
					government_legitimacy >= 20
				}
			}
			NOT = { has_variable = gbr_end_rationing_early_button_pressed }
		}
	}
	
	ai_chance = {
		base = 0
		modifier = { trigger = { var:gbr_rationing_number_of_goods <= 10 } add = 0.1 }
		modifier = { trigger = { var:gbr_rationing_number_of_goods <= 9 } add = 0.1 }
		modifier = { trigger = { var:gbr_rationing_number_of_goods <= 8 } add = 0.1 }
		modifier = { trigger = { var:gbr_rationing_number_of_goods <= 7 } add = 0.1 }
		modifier = { trigger = { var:gbr_rationing_number_of_goods <= 6 } add = 0.1 }
		modifier = { trigger = { var:gbr_rationing_number_of_goods <= 5 } add = 0.1 }
		modifier = { trigger = { var:gbr_rationing_number_of_goods <= 4 } add = 0.1 }
		modifier = { trigger = { var:gbr_rationing_number_of_goods <= 3 } add = 0.1 }
		modifier = { trigger = { var:gbr_rationing_number_of_goods <= 2 } add = 0.1 }
		modifier = { trigger = { var:gbr_rationing_number_of_goods <= 1 } add = 0.1 }
		modifier = { trigger = { government_legitimacy >= 60 } add = 0.25 }
		modifier = { trigger = { government_legitimacy >= 70 } add = 0.25 }
		modifier = { trigger = { government_legitimacy >= 80 } add = 0.25 }
		modifier = { trigger = { government_legitimacy >= 90 } add = 0.25 }
		modifier = { trigger = { has_healthy_economy = yes } add = 0.5 }
		modifier = { trigger = { taking_loans = no } add = 0.5 }
	}
	
	effect = {
		hidden_effect = {
			set_variable = gbr_end_rationing_early_button_pressed
			set_variable = gbr_ended_rationing_early
			set_variable = gbr_rationing_over
		}
	}
}
