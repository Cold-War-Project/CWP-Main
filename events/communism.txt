namespace = communism
# Rising socialist leader
communism.7 = {
	type = country_event
	placement = root
	title = communism.7.t
	desc = communism.7.d
	flavor = communism.7.f
	duration = 3
	cooldown = {
		months = normal_modifier_time
	}
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	event_image = {
		video = "votp_agitator_speaking"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	trigger = {
		NOT = {
			has_law = law_type:law_council_elective
		}
		any_interest_group = {
			is_marginal = no
			any_scope_character = {
				OR = {
					has_role = politician
					has_role = agitator
				}
				NOT = {
					has_modifier = popular_socialist_leader
				}
				has_socialist_ideology = yes
			}
		}
	}
	immediate = {
		random_interest_group = {
			limit = {
				is_marginal = no
				any_scope_character = {
					OR = {
						has_role = politician
						has_role = agitator
					}
					NOT = {
						has_modifier = popular_socialist_leader
					}
					has_socialist_ideology = yes
				}
			}
			save_scope_as = popular_socialist_ig
			random_scope_character = {
				limit = {
					OR = {
						has_role = politician
						has_role = agitator
					}
					NOT = {
						has_modifier = popular_socialist_leader
					}
					has_socialist_ideology = yes
				}
				save_scope_as = popular_socialist_leader
			}
		}
	}
	option = {
		name = communism.7.a
		trigger = {
			scope:popular_socialist_ig = {
				is_in_government = yes
			}
		}
		default_option = yes
		scope:popular_socialist_leader = {
			add_modifier = {
				name = popular_socialist_leader
				months = short_modifier_time
			}
		}
	}
	option = {
		name = communism.7.b
		trigger = {
			scope:popular_socialist_ig = {
				is_in_government = no
			}
		}
		default_option = yes
		scope:popular_socialist_leader = {
			add_modifier = {
				name = popular_socialist_leader
				months = short_modifier_time
			}
		}
		random_scope_state = {
			limit = {
				is_incorporated = yes
			}
			add_radicals_in_state = {
				value = large_radicals
				strata = poor
			}
		}
	}
}

# The Red Army
communism.8 = {
	type = country_event
	placement = root
	title = communism.8.t
	desc = communism.8.d
	flavor = communism.8.f
	duration = 3
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	event_image = {
		video = "unspecific_military_parade"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	trigger = {
		has_law = law_type:law_council_elective
		NOT = {
			has_variable = red_army_var
		}
		ig:ig_securocrats = {
			leader = {
				has_socialist_ideology = yes
			}
		}
	}
	immediate = {
		ig:ig_securocrats = {
			save_scope_as = red_army_ig
			leader = {
				save_scope_as = red_army_general
			}
		}
		set_variable = red_army_var
	}
	option = {
		name = communism.8.a
		ig:ig_securocrats = {
			custom_tooltip = red_army_tt
			hidden_effect = {
				set_interest_group_name = ig_red_army
			}
			add_ideology = ideology_vanguardist
		}
	}
}
