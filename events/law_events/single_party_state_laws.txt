# Advance Event
single_party_state.1 = {
	type = country_event
	placement = ROOT
	
	title = single_party_state.1.t
	desc = single_party_state.1.d
	flavor = single_party_state.1.f

	category = enactment
	
	event_image = {
		video = "europenorthamerica_political_extremism"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"
	
	duration = 3
	
	cooldown = { months = normal_modifier_time }
	
	trigger = {
		is_enacting_law = law_type:law_anocracy
		ruler.interest_group = {
			has_party = yes
		}
		any_interest_group = {
			law_stance = {
				law = law_type:law_anocracy
				value >= neutral
			}	
			NOT = {
				this = ruler.interest_group
				this.party = ruler.interest_group.party
				OR = {
					AND = {
						owner.ruler = {
							has_fascist_ideology = yes
						}
						this.leader = {
							has_socialist_ideology = yes
						}
					}
					AND = {
						owner.ruler = {
							has_socialist_ideology = yes
						}
						this.leader = {
							has_fascist_ideology = yes
						}
					}
				}
			}
		}
	}

	immediate = {
		currently_enacting_law = {
			save_scope_as = current_law_scope
		}
		ruler.interest_group.party = {
			save_scope_as = ruling_party
		}
		random_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_anocracy
					value >= neutral
				}	
				NOT = {
					this = ruler.interest_group
					this.party = ruler.interest_group.party
					OR = {
						AND = {
							owner.ruler = {
								has_fascist_ideology = yes
							}
							this.leader = {
								has_socialist_ideology = yes
							}
						}
						AND = {
							owner.ruler = {
								has_socialist_ideology = yes
							}
							this.leader = {
								has_fascist_ideology = yes
							}
						}
					}
				}
			}
			save_scope_as = party_switch_ig
		}
	}

	option = {
		name = single_party_state.1.a
		trigger = {
			enactment_phase < 2
		}
		highlighted_option = yes
		add_enactment_phase = 1
		add_loyalists = {
			value = small_radicals
			interest_group = scope:party_switch_ig
		}
	}

	option = {
		name = single_party_state.1.b
		default_option = yes
		scope:party_switch_ig = {
			add_modifier = {
				name = modifier_model_ig_in_system
				months = normal_modifier_time
			}
		}
		add_enactment_modifier = {
			name = modifier_parties_supporting_sps
			multiplier = 1
		}
		add_loyalists = {
			value = medium_radicals
			interest_group = scope:party_switch_ig
		}
	}
}
