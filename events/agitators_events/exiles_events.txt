					interest_group_supports_political_movement = no
				}
			}
			this.interest_group = {
				save_scope_as = agitator_ig
			}
			save_scope_as = agitator_scope

		name = exiles_events.1.a

exiles_events.3 = {
	type = country_event
	placement = root

	duration = 3

	title = exiles_events.3.t
	desc = exiles_events.3.d
	flavor = exiles_events.3.f

	event_image = {
		video = "votp_realist_household"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	cooldown = { months = normal_modifier_time }

	trigger = { #If an interest group has a character in the exile pool and supports a movement, this fires.
		empty_agitator_slots >= 1
		any_political_movement = {
			any_supporting_interest_group = {
				interest_group_supports_political_movement = yes
				type = {
					save_temporary_scope_as = ex_ig_type
				}
			}
			this.desired_law = {
				save_temporary_scope_as = movement_law
			}
		}
		any_character_in_exile_pool = {
			root = this.home_country
			this.interest_group_type = scope:ex_ig_type
			law_stance = {
				law = scope:movement_law
				value > neutral
			}
		}
	}

	cancellation_trigger = {
		NOT = {
			exists = scope:exile_scope
		}
	}
	
	immediate = {
		random_interest_group = {
			limit = {
				interest_group_supports_political_movement = yes
				any_character_in_exile_pool = {
					this.interest_group_type = prev.type
				}
			}
			type = {
				save_temporary_scope_as = ex_ig_type
			}
			save_scope_as = exile_ig
		}
		random_political_movement = {
			limit = {
				any_supporting_interest_group = {
					is_same_interest_group_type = scope:exile_ig
				}
			}
			this.desired_law = {
				save_scope_as = movement_law
			}
			save_scope_as = movement_scope
		}
		random_character_in_exile_pool = {
			limit = {
				root = this.home_country
				this.interest_group_type = scope:ex_ig_type
				law_stance = {
					law = scope:movement_law
					value > neutral
				}
			}
			save_scope_as = exile_scope
		}
	}
	
	option = {
		name = exiles_events.3.a 
		default_option = yes
		scope:exile_scope = {
			transfer_character = root
			hidden_effect = {
				remove_modifier = in_exile_modifier
			}
		}
		scope:movement_scope = {
			add_modifier = {
				name = modifier_agitator_invited
				months = normal_modifier_time
			}
		}
	}

	option = {
		name = exiles_events.3.b
		scope:movement_scope = {
			add_modifier = {
				name = modifier_agitator_refused
				months = normal_modifier_time
			}
		}
		scope:exile_scope = {
			add_modifier = {
				name = modifier_agitator_coward
				months = normal_modifier_time
			}
		}
	}
}
