# Exile attempts to return to break law passage deadlock
exiles_more_events.2 = {
	type = country_event
	placement = root
	duration = 3
	title = exiles_more_events.2.t
	desc = exiles_more_events.2.d
	flavor = exiles_more_events.2.f
	event_image = {
		video = "votp_gunboat_diplomacy"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	cooldown = {
		months = normal_modifier_time
	}
	trigger = {
		enacting_any_law = yes
		OR = {
			AND = {
				any_interest_group = {
					law_stance = {
						law = owner.currently_enacting_law.type
						value = strongly_approve
					}
					leader = {
						NOT = {
							has_modifier = ideological_victor
						}
					}
					save_temporary_scope_as = ex_ig
				}
				any_character_in_exile_pool = {
					home_country = ROOT
					ig_type_matches_former = yes
					law_stance = {
						law = root.currently_enacting_law.type
						value = strongly_approve
					}
				}
			}
			AND = {
				any_interest_group = {
					law_stance = {
						law = owner.currently_enacting_law.type
						value = strongly_disapprove
					}
					leader = {
						NOT = {
							has_modifier = ideological_victor
						}
					}
					save_temporary_scope_as = ex_ig
				}
				any_character_in_exile_pool = {
					home_country = ROOT
					ig_type_matches_former = yes
					law_stance = {
						law = root.currently_enacting_law.type
						value = strongly_disapprove
					}
				}
			}
		}
	}
	immediate = {
		currently_enacting_law = {
			save_scope_as = interventionist_exile_law_scope
		}
		random_character_in_exile_pool = {
			limit = {
				home_country = root
				OR = {
					AND = {
						law_stance = {
							law = root.currently_enacting_law.type
							value = strongly_approve
						}
						interest_group_type = {
							save_temporary_scope_as = exile_ig_type
						}
						root = {
							any_interest_group = {
								this.type = scope:exile_ig_type
								law_stance = {
									law = owner.currently_enacting_law.type
									value = strongly_approve
								}
							}
						}
					}
					AND = {
						law_stance = {
							law = root.currently_enacting_law.type
							value = strongly_disapprove
						}
						interest_group_type = {
							save_temporary_scope_as = exile_ig_type
						}
						root = {
							any_interest_group = {
								this.type = scope:exile_ig_type
								law_stance = {
									law = owner.currently_enacting_law.type
									value = strongly_disapprove
								}
							}
						}
					}
				}
			}
			save_temporary_scope_as = ex_ig
			save_scope_as = interventionist_exile_scope
		}
		random_interest_group = {
			limit = {
				ig_type_matches_former = yes
			}
			leader = {
				save_scope_as = interventionist_exile_ig_leader
			}
			save_scope_as = interventionist_exile_ig
		}
	}
	cancellation_trigger = {
		scope:interventionist_exile_scope = {
			is_in_exile_pool = no
		}
	}
	option = {
		name = exiles_more_events.2.a
		default_option = yes
		scope:interventionist_exile_scope = {
			transfer_character = ROOT
			hidden_effect = {
				remove_modifier = in_exile_modifier
			}
		}
	}
	option = {
		name = exiles_more_events.2.b
		if = {
			limit = {
				scope:interventionist_exile_ig = {
					law_stance = {
						law = owner.currently_enacting_law.type
						value = strongly_approve
					}
				}
			}
			add_enactment_modifier = {
				name = positive_exile_intervention
			}
		}
		else_if = {
			limit = {
				scope:interventionist_exile_ig = {
					law_stance = {
						law = owner.currently_enacting_law.type
						value = strongly_disapprove
					}
				}
			}
			add_enactment_modifier = {
				name = negative_exile_intervention
			}
		}
	}
	option = {
		name = exiles_more_events.2.c
		highlighted_option = yes
		trigger = {
			custom_tooltip = {
				text = exiles_more_events.2.c.tt
				OR = {
					AND = {
						scope:interventionist_exile_scope = {
							has_liberal_ideology = yes
						}
						scope:interventionist_exile_ig = {
							leader = {
								has_liberal_ideology = no
							}
						}
					}
					AND = {
						scope:interventionist_exile_scope = {
							has_progressive_ideology = yes
						}
						scope:interventionist_exile_ig = {
							leader = {
								has_progressive_ideology = no
							}
						}
					}
					AND = {
						scope:interventionist_exile_scope = {
							has_socialist_ideology = yes
						}
						scope:interventionist_exile_ig = {
							leader = {
								has_socialist_ideology = no
							}
						}
					}
					AND = {
						scope:interventionist_exile_scope = {
							has_reactionary_ideology = yes
						}
						scope:interventionist_exile_ig = {
							leader = {
								has_reactionary_ideology = no
							}
						}
					}
					AND = {
						scope:interventionist_exile_scope = {
							has_conservative_ideology = yes
						}
						scope:interventionist_exile_ig = {
							leader = {
								has_conservative_ideology = no
							}
						}
					}
				}
			}
		}
		scope:interventionist_exile_ig = {

# Agitator has no domestic support and sullenly goes into self-exile
exiles_more_events.3 = {
	type = country_event
	placement = root
	duration = 3
	title = exiles_more_events.3.t
	desc = exiles_more_events.3.d
	flavor = exiles_more_events.3.f
	gui_window = event_window_1char_tabloid
	left_icon = scope:the_loneliest_little_agitator
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	cooldown = {
		months = normal_modifier_time
	}
	trigger = {
		any_scope_character = {
			has_role = agitator
			political_movement = {
				political_movement_support < 15
			}
		}
	}
	immediate = {
		random_scope_character = {
			limit = {
				has_role = agitator
				political_movement = {
					political_movement_support < 15
				}
			}
			save_scope_as = the_loneliest_little_agitator
			interest_group = {
				save_scope_as = the_loneliest_little_agitator_ig
			}
		}
	}
	option = {
		name = exiles_more_events.3.a
		default_option = yes
		scope:the_loneliest_little_agitator = {
			exile_character = yes
		}
	}
	option = {
		# Perhaps it's time for a little study and reflection?
		name = exiles_more_events.3.b
		highlighted_option = yes
		trigger = {
			scope:the_loneliest_little_agitator = {
				has_trait = erudite
			}
		}
		scope:the_loneliest_little_agitator = {
			add_modifier = {
				name = inspired_by_studies
				months = normal_modifier_time
			}
		}
	}
	option = {
		# time for something radical
		name = exiles_more_events.3.c
		highlighted_option = yes
		trigger = {
			scope:the_loneliest_little_agitator = {
				not = {
					has_trait = cautious
				}
			}
		}
		random_list = {
			1 = {
				scope:the_loneliest_little_agitator = {
					political_movement = {
						add_modifier = {
							name = radical_new_approach
							months = normal_modifier_time
						}
					}
				}
			}
			1 = {
				scope:the_loneliest_little_agitator = {
					political_movement = {
						add_modifier = {
							name = failed_relaunch
							months = normal_modifier_time
						}
					}
				}
			}
			1 = {
				scope:the_loneliest_little_agitator = {
					political_movement = {
						add_modifier = {
							name = all_worked_up
							months = normal_modifier_time
						}
					}
				}
			}
		}
	}
	option = {
		# There's no need for anything drastic like exile!
		name = exiles_more_events.3.dd
		highlighted_option = yes
		trigger = {
			scope:the_loneliest_little_agitator = {
				has_trait = cautious
			}
		}
		scope:the_loneliest_little_agitator = {
			political_movement = {
				add_modifier = {
					name = deliberate_deliberations
					months = normal_modifier_time
				}
			}
		}
	}
}
