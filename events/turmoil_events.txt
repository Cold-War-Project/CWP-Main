namespace = turmoil_events

#Low turmoils
turmoil_events.1 = { #Striking
	type = country_event
	placement = scope:turmoil_striking_state

	title = turmoil_events.1.t
	desc = turmoil_events.1.d
	flavor = turmoil_events.1.f

	event_image = {
		video = "unspecific_trains"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3

	trigger = {
		NOT = { has_variable = turmoil_1_cooldown }
		any_state = {
			owner = ROOT
			turmoil >= 0.25
			turmoil <= 0.75
		}
	}
	immediate = {
		set_variable = {
			name = turmoil_1_cooldown
			months = 3
		}
		random_state = {
			limit = {
				owner = ROOT
				turmoil >= 0.25
				turmoil <= 0.75
			}
			save_scope_as = turmoil_striking_state
		}
	}
	option = {
		name = turmoil_events.1.a
		scope:turmoil_striking_state = {
			add_modifier = {
				name = modifier_anti_striking_measures
				months = 4
			}
			add_radicals_in_state = {
   				value = medium_radicals
   				strata = poor
			}
		}
	}
	option = {
		name = turmoil_events.1.b
		default_option = yes
		scope:turmoil_striking_state = {
			add_modifier = {
				name = modifier_turmoil_striking
				months = 4
			}
			add_radicals_in_state = {
   				value = small_radicals
   				strata = poor
			}
		}
	}
	option = {
		name = turmoil_events.1.c
		default_option = yes
		scope:turmoil_striking_state = {
			add_modifier = {
				name = modifier_turmoil_strike_aid
				months = 4
			}
		}
		add_modifier = {
			name = modifier_turmoil_strike_aid_cost
			months = 4
		}
	}
}
