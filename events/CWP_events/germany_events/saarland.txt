namespace = saarland

saarland.1 = { # France creates the Saar Protectorate

	type = country_event
	title = saarland.1.t
	desc = saarland.1.d
	flavor = saarland.1.f
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	event_image = {
		texture = "/gfx/interface/event_pictures/german_unification_bundestag.dds"
	}

	duration = 3

	trigger = {
		AND = {
			exists = c:FRA
			this = c:FRA
			exists = s:STATE_RHINELAND.region_state:FOZ
			c:FOZ = {is_subject_of = c:FRA}
			NOT = {exists = c:SAA}
			game_date >= 1946.2.16
		}
		NOT = {
			has_variable = saarland1_event_fired
		}
	}
	immediate = {
		set_variable = saarland1_event_fired
	}
	option = {
		name = saarland.1.a
		default_option = yes
		ai_chance = {
			base = 75
		}
		c:FOZ = {
			create_country = {
				tag = SAA
				origin = this
				province = p:x904020
			}
		}
		if = {
			limit = { exists = c:SAA }
			create_diplomatic_pact = {
				country = c:SAA
				type = puppet
			}
			c:SAA = { 
				effect_currency_frf = yes
				onaction_monetary_variables_creation = yes
			}
		}
		custom_tooltip = {
			text = saarland.1.a.tt
		}
	}
	option = {
		name = saarland.1.b
		ai_chance = {
			base = 0
		}
		c:FOZ = {
			create_country = {
				tag = SAA
				origin = this
				province = p:x904020
			}
		}
		if = {
			limit = { exists = c:SAA }
			create_diplomatic_pact = {
				country = c:SAA
				type = puppet
			}
			c:SAA = { 
				effect_currency_frf = yes
				onaction_monetary_variables_creation = yes
			}
			play_as = c:SAA
		}
		custom_tooltip = {
			text = saarland.1.a.tt
		}
	}
	option = {
		name = saarland.1.c
		ai_chance = {
			base = 25
		}
	}
}
