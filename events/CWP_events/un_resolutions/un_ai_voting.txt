namespace = un_ai_voting
un_ai_voting.1 = {
	# AI voting mechanism
	type = country_event
	placement = ROOT
	trigger = {
		is_ai = yes
		has_global_variable = pending_resolution
		OR = {
			has_variable = un_member
			has_variable = unsc_member
		}
		nor = {
			has_variable = un_alreadyvoted
			has_global_variable = vetoed
		}
	}
	immediate = {
		every_country = {
			limit = {
				has_variable = asking_un_intervention
			}
			save_scope_as = asking_un_intervention_country
		}
		every_country = {
			limit = {
				has_variable = un_intervention_enemy_country
			}
			save_scope_as = un_intervention_enemy_country
		}
	}
	# TODO - AI SCRIPTING
	option = {
		default_option = yes
		name = un_ai_voting.1.a
		effect_ai_vote_yes = yes
		ai_chance = {
			base = 0			# Declaration of human rights
			modifier = {
				trigger = {
					and = {
						has_journal_entry = je_korean_resolution
						is_authoritarian = no
					}
				}
				add = 50
			}
			# Declaration of human rights
			modifier = {
				trigger = {
					and = {
						has_journal_entry = je_declaration_of_human_rights
						is_authoritarian = no
					}
				}
				add = 50
			}
			# Resolution on Decolonization
			modifier = {
				trigger = {
					and = {
						has_journal_entry = je_decolonization_resolution
						has_anti_imperialist_policy = yes
					}
				}
				add = 50
			}
			modifier = {
				trigger = {
					and = {
						has_journal_entry = je_decolonization_resolution
						country_rank >= rank_value:superpower
					}
				}
				add = 100
			}
			# UNCLOS 1
			modifier = {
				trigger = {
					and = {
						has_journal_entry = je_unclos1
						any_scope_state = {
							is_coastal = yes
						}
					}
				}
				add = 50
			}
			# UN Intervention
			modifier = {
				trigger = {
					and = {
						has_journal_entry = je_un_intervention
						scope:asking_un_intervention_country = {
							relations:root > relations_threshold:cordial
						}
					}
				}
				add = 50
			}
			modifier = {
				trigger = {
					and = {
						has_journal_entry = je_un_intervention
						scope:un_intervention_enemy_country = {
							relations:root < relations_threshold:hostile
						}
					}
				}
				add = 50
			}
		}
	}
	option = {
		name = un_ai_voting.1.b
		effect_ai_vote_no = yes
		ai_chance = {
			base = 0
			modifier = {
				trigger = {
					and = {
						has_journal_entry = je_korean_resolution
						is_authoritarian = yes
					}
				}
				add = 60
			}
			# UNCLOS 1
			modifier = {
				trigger = {
					and = {
						has_journal_entry = je_unclos1
					}
				}
				add = 55
			}
			# Resolution on Decolonization
			modifier = {
				trigger = {
					and = {
						has_journal_entry = je_decolonization_resolution
						has_imperialist_policy = yes
					}
				}
				add = 50
			}
			# UN Intervention
			modifier = {
				trigger = {
					and = {
						has_journal_entry = je_un_intervention
						scope:asking_un_intervention_country = {
							relations:root < relations_threshold:neutral
						}
					}
				}

					and = {
						has_journal_entry = je_un_intervention
						scope:asking_un_intervention_country = {
							relations:root < relations_threshold:cold

						scope:asking_un_intervention_country = {

un_ai_voting.2 = {
	# AI Application voting mechanism
	type = country_event
	placement = ROOT
	trigger = {
		is_ai = yes
		OR = {
			has_variable = un_member
			has_variable = unsc_member
		}
		not = {
			has_variable = has_voted_on_application
		}
		any_country = {
			has_variable = application_progress_un
		}
	}
	immediate = {
		random_country = {
			limit = {
				has_variable = application_progress_un
			}
			save_scope_as = applicant_country_scope
		}
	}
	# TODO - AI SCRIPTING
	option = {
		default_option = yes
		name = un_ai_voting.2.a
		set_variable = {
			name = has_voted_on_application
			days = 50
		}
		if = {
			limit = {
				or = {
					supporting_countries <= un_memberstates
					not = {
						exists = var:un_supporting_list_size
					}
				}
			}
			scope:applicant_country_scope = {
				if = {
					limit = {
						not = {
							is_target_in_variable_list = {
								name = un_supporting
								target = ROOT
							}
						}
					}
					change_variable = {
						name = un_supporting_list_size
						add = 1
					}
					add_to_variable_list = {
						name = un_supporting
						target = ROOT
					}
				}
				if = {
					limit = {
						is_target_in_variable_list = {
							name = un_rejecting
							target = ROOT
						}
					}
					change_variable = {
						name = un_supporting_list_size
						add = 1
					}
				}
				remove_list_variable = {
					name = un_rejecting
					target = ROOT
				}
				remove_list_variable = {
					name = un_abstaining
					target = ROOT
				}
			}
		}
		ai_chance = {
			base = 30
			modifier = {
				trigger = {
					scope:applicant_country_scope = {
						relations:root >= relations_threshold:cordial
					}
				}
				add = 100
			}
		}
	}
	option = {
		name = un_ai_voting.2.b
		set_variable = {
			name = has_voted_on_application
			days = 50
		}
		if = {
			limit = {
				or = {
					supporting_countries <= un_memberstates
					not = {
						exists = var:un_supporting_list_size
					}
				}
			}
			scope:applicant_country_scope = {
				add_to_variable_list = {
					name = un_abstaining
					target = ROOT
				}
				if = {
					limit = {
						is_target_in_variable_list = {
							name = un_supporting
							target = ROOT
						}
					}
					change_variable = {
						name = un_supporting_list_size
						subtract = 1
					}
				}
				if = {
					limit = {
						is_target_in_variable_list = {
							name = un_rejecting
							target = ROOT
						}
					}
					change_variable = {
						name = un_supporting_list_size
						add = 1
					}
				}
				remove_list_variable = {
					name = un_supporting
					target = ROOT
				}
				remove_list_variable = {
					name = un_rejecting
					target = ROOT
				}
			}
		}
		ai_chance = {
			base = 10
			modifier = {
				trigger = {
					scope:applicant_country_scope = {
						relations:root = relations_threshold:neutral
					}
				}
				add = 5
			}
		}
	}
	option = {
		name = un_ai_voting.2.c
		set_variable = {
			name = has_voted_on_application
			days = 50
		}
		if = {
			limit = {
				or = {
					supporting_countries <= un_memberstates
					not = {
						exists = var:un_supporting_list_size
					}
				}
			}
			scope:applicant_country_scope = {
				if = {
					limit = {
						not = {
							is_target_in_variable_list = {
								name = un_rejecting
								target = ROOT
							}
						}
					}
					change_variable = {
						name = un_supporting_list_size
						subtract = 1
					}
					add_to_variable_list = {
						name = un_rejecting
						target = ROOT
					}
				}
				if = {
					limit = {
						is_target_in_variable_list = {
							name = un_supporting
							target = ROOT
						}
					}
					change_variable = {
						name = un_supporting_list_size
						subtract = 1
					}
				}
				remove_list_variable = {
					name = un_supporting
					target = ROOT
				}
				remove_list_variable = {
					name = un_abstaining
					target = ROOT
				}
			}
		}
		ai_chance = {
			base = 5

				trigger = {
					scope:applicant_country_scope = {
						relations:root < relations_threshold:neutral
					}
				}
				add = 30
			}

				trigger = {
					scope:applicant_country_scope = {
						relations:root <= relations_threshold:hostile
					}
				}
				add = 30
