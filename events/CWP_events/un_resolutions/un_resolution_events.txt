un_resolution_events.100 = {
	# AI initializes vote on Decolonization
	type = country_event
	hidden = yes
	trigger = {
		or = {
			has_variable = unsc_member
			has_variable = un_member
		}
		is_player = no
		country_rank = rank_value:superpower
		game_date >= 1956.1.1
		NOR = {
			has_global_variable = pending_resolution
			has_global_variable = decolonization_resolutioncomplete
			has_global_variable = decolonization_resolutionfail
		}
	}
	immediate = {
		activate_un_decolonization_je = yes
	}
}
