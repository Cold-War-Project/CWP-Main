						has_global_variable = newfoundland_player

		}

newfoundland_referendum.2 = { # Where are we sending delegates? London? Ottawa? Both?
	type = country_event
	title = newfoundland_referendum.2.title
	desc = newfoundland_referendum.2.desc
	flavor = newfoundland_referendum.2.flavor
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	event_image = {
		texture = "gfx/interface/event_pictures/newfoundland_referendum/national_convention_newfoundland.dds"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/middleeast/courtroom_upheaval"
	
	duration = 2
	
	trigger = {
		game_date >= 1946.5.30
		this = c:NEW
		NOT = { has_global_variable = where_are_we_sending_delegates }
		has_global_variable = newfoundland_national_convention_start
	}
	
	immediate = {
		set_global_variable = where_are_we_sending_delegates
	}
	
	option = {
		default_option = yes
		name = newfoundland_referendum.2.a # Delegation to both London & Ottawa
		ai_chance = {
			base = 60	# 60%
			modifier = {	# If both UK & Canada are player controlled, then both should have a say
				trigger = {
					AND = {
						has_global_variable = canada_player
						has_global_variable = uk_player
					}
				}
				add = 40	# 100%
			}
			modifier = {	# If only one of the two, UK or Canada, are player controlled, then severely reduce the chance of delagates being sent to the non-player controlled country
				trigger = {
					OR = {
						AND = {
							NOT = { has_global_variable = canada_player }
							has_global_variable = uk_player
						}
						AND = {
							NOT = { has_global_variable = uk_player }
							has_global_variable = canada_player
						}
					}
				}
				add = -40	# 60% - 40% = 20%
			}
		}
		custom_tooltip = {
			text = newfoundland_referendum.2.a.tt
		}
		hidden_effect = {
			set_global_variable = london_and_ottawa_delegation
		}
	}
	option = {
		name = newfoundland_referendum.2.b # Delegation to London only
		ai_chance = {
			base = 20	# 20%
			modifier = {	# UK player most likely to get sole influence, but a chance remains for Canada to have their say
				trigger = {
					NOT = { has_global_variable = canada_player }
					has_global_variable = uk_player
				}
				add = 60	# 80%
			}
			modifier = {	# Canada player should be able to have influence, so effectively remove the possibility of Canada not having influence
				trigger = {
					has_global_variable = canada_player
					NOT = { has_global_variable = uk_player }
				}
				add = -20	# 0%
			}
		}
		custom_tooltip = {
			text = newfoundland_referendum.2.b.tt
		}
		hidden_effect = {
			set_global_variable = london_delegation_only
		}
	}
	option = {
		name = newfoundland_referendum.2.c # Delegation to Ottawa only
		ai_chance = {
			base = 20	# 20%
			modifier = {	# Canada player most likely to get sole influence, but a chance remains for UK to have their say
				trigger = {
					NOT = { has_global_variable = uk_player }
					has_global_variable = canada_player
				}
				add = 60	# 80%
			}
			modifier = {	# UK player should be able to have influence, so effectively remove the possibility of UK not having influence
				trigger = {
					has_global_variable = uk_player
					NOT = { has_global_variable = canada_player }
				}
				add = -20	# 0%
			}

		}
		custom_tooltip = {
			text = newfoundland_referendum.2.c.tt
		}
		hidden_effect = {
			set_global_variable = ottawa_delegation_only
		}
	}

newfoundland_referendum.3 = { # London Delegation

	on_opened_soundeffect = "event:/SFX/Events/middleeast/courtroom_upheaval"
	
	duration = 1
	
	trigger = {
		game_date >= 1947.4.30
		this = c:GBR
		NOR = {
			has_global_variable = london_delegation_started
			has_global_variable = london_delegation_finished
		}
		OR = {
			has_global_variable = london_and_ottawa_delegation
			has_global_variable = london_delegation_only
		}
	}
	
	immediate = {
		set_global_variable = london_delegation_started
	}
	
	option = {
		default_option = yes
		name = newfoundland_referendum.3.a # Join with Canada
		ai_chance = {
			base = 33	# 33%
			modifier = {	# If there is no Newfoundland player but there is a Canada player, we assume the Canada player would like to confederate with Newfoundland
				trigger = {
					NOT = { has_global_variable = newfoundland_player }
					has_global_variable = canada_player
				}
				add = 67	# 100%
			}
			modifier = {	# If there is a Newfoundland player, we don't want the UK to influence the referendum, so we effectively remove the possibility
				trigger = {
					has_global_variable = newfoundland_player
				}
				add = -33	# 0%
			}
		}
		custom_tooltip = {
			text = newfoundland_referendum.3.a.tt
		}
		hidden_effect = {
			set_global_variable = london_delegation_finished
			set_global_variable = london_choose_confederation
		}
	}
	option = {
		name = newfoundland_referendum.3.b # We have no preference
		ai_chance = {
			base = 34	# 34%
			modifier = {	# Give the Newfoundland player control over the referendum outcome without being influenced
				trigger = {
					has_global_variable = newfoundland_player
				}
				add = 66	# 100%
			}
			modifier = {	# If there is no Newfoundland player but there is a Canada player, we assume the Canada player would like to confederate with Newfoundland so we effectively remove this option
				trigger = {
					NOT = { has_global_variable = newfoundland_player }
					has_global_variable = canada_player
				}
				add = -34	# 0%
			}
		}
		custom_tooltip = {
			text = newfoundland_referendum.3.b.tt
		}
		hidden_effect = {
			set_global_variable = london_delegation_finished
			set_global_variable = london_no_preference
		}
	}
	option = {
		name = newfoundland_referendum.3.c # Become a Dominion
		ai_chance = {
			base = 33	# 33%
			modifier = {	# If there is a Canada player or Newfoundland player, it is unlikely they would want to remain a British subject even if it is an upgrade in subject type
				trigger = {
					OR = {
						has_global_variable = newfoundland_player
						has_global_variable = canada_player
					}
				}
				add = -33	# 0%
			}
		}
		custom_tooltip = {

		}
		hidden_effect = {
			set_global_variable = london_delegation_finished
			set_global_variable = london_choose_dominion
		}
	}
}

newfoundland_referendum.998 = { # Determining whether the first referendum will have an independence option
	type = country_event
	hidden = yes
	trigger = {
		OR = {
			AND = {
				has_global_variable = london_and_ottawa_delegation 
				OR = {
					has_global_variable = london_no_newfoundland_independence
					has_global_variable = london_yes_newfoundland_independence
				}
				OR = {
					has_global_variable = ottawa_no_newfoundland_independence
					has_global_variable = ottawa_yes_newfoundland_independence
				}
			}
			AND = {
				has_global_variable = london_delegation_only
				OR = {
					has_global_variable = london_no_newfoundland_independence
					has_global_variable = london_yes_newfoundland_independence
				}
			}
			AND = {
				has_global_variable = ottawa_delegation_only
				OR = {
					has_global_variable = ottawa_no_newfoundland_independence
					has_global_variable = ottawa_yes_newfoundland_independence
				}
			}
		}
		NOT = { has_global_variable = newfoundland_first_referendum_started }
	}
	immediate = {
		if = {	# Both UK and Canada received Newfoundland delegations and said no to Newfoundland independence
			limit = {
				has_global_variable = london_and_ottawa_delegation
				has_global_variable = ottawa_no_newfoundland_independence
				has_global_variable = london_no_newfoundland_independence
				NOT = { has_global_variable = newfoundland_player }	# If there is a Newfoundland player, we want them to have the independence option
			}
			set_global_variable = newfoundland_first_referendum_no_independence
		}
		if = {	# Both UK and Canada received Newfoundland delegations, but the UK said yes to Newfoundland independence while Canada said no
			limit = {
				has_global_variable = london_and_ottawa_delegation
				has_global_variable = ottawa_no_newfoundland_independence
				has_global_variable = london_yes_newfoundland_independence
				NOT = { has_global_variable = newfoundland_player }	# If there is a Newfoundland player, we want them to have the independence option
			}
			random_list = {	# UK's prefered option is more likely due to there being a power imbalance between the UK and Canada
				40 = {	# 40%
					set_global_variable = newfoundland_first_referendum_no_independence
				}
				60 = {	# 60%
					set_global_variable = newfoundland_first_referendum_yes_independence
				}
			}
		}
		if = {	# Both UK and Canada received Newfoundland delegations, but Canada said yes to Newfoundland independence while the UK said no
			limit = {
				has_global_variable = london_and_ottawa_delegation
				has_global_variable = ottawa_no_newfoundland_independence
				has_global_variable = london_yes_newfoundland_independence
				NOT = { has_global_variable = newfoundland_player }	# If there is a Newfoundland player, we want them to have the independence option
			}
			random_list = {	# UK's prefered option is more likely due to there being a power imbalance between the UK and Canada
				60 = {	# 60%
					set_global_variable = newfoundland_first_referendum_no_independence
				}
				40 = {	# 40%
					set_global_variable = newfoundland_first_referendum_yes_independence
				}
			}
		}
		if = {	# If both the UK and Canada said yes to Newfoundland having the option of independence or there is a Newfoundland player, then choose the referendum that has the independence option
			limit = {
				OR = {
					AND = {
						has_global_variable = london_and_ottawa_delegation
						has_global_variable = ottawa_yes_newfoundland_independence
						has_global_variable = london_yes_newfoundland_independence
					}
					has_global_variable = newfoundland_player
				}
			}
			set_global_variable = newfoundland_first_referendum_yes_independence
		}
		if = {	# If only one delegation was sent and the receiving country says yes to the possibility of Newfoundland independence or there is a Newfoundland player
			limit = {
				OR = {
					AND = {
						has_global_variable = london_delegation_only
						has_global_variable = london_yes_newfoundland_independence
					}
					AND = {
						has_global_variable = ottawa_delegation_only
						has_global_variable = ottawa_yes_newfoundland_independence
					}
					has_global_variable = newfoundland_player
				}
			}
			set_global_variable = newfoundland_first_referendum_yes_independence
		}
		if = {	# If only one delegation was sent and the receiving country says no to the possibility of Newfoundland independence and there isn't a Newfoundland player
			limit = {
				OR = {
					AND = {
						has_global_variable = london_delegation_only
						has_global_variable = london_no_newfoundland_independence
					}
					AND = {
						has_global_variable = ottawa_delegation_only
						has_global_variable = ottawa_no_newfoundland_independence
					}
				}
				NOT = { has_global_variable = newfoundland_player }
			}
			set_global_variable = newfoundland_first_referendum_no_independence
		}
	}
}
