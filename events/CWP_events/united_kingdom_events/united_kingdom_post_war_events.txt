united_kingdom_post_war_events.101 = {
	# Pound Sterling Devaluation
	# Determine what the new value of the pound sterling will be when devaluing it
	type = country_event
	title = united_kingdom_post_war_events.101.title
	desc = united_kingdom_post_war_events.101.desc
	flavor = united_kingdom_post_war_events.101.flavor
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	event_image = { video = "gfx/event_pictures/middleeast_courtroom_upheaval.bk2" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/middleeast/courtroom_upheaval"
	
	duration = 1
	
	immediate = { }
	
	option = {
		name = united_kingdom_post_war_events.101.a
		default_option = yes
		if = {
			limit = {
				var:uk_rationing_inflation_rate >= 10
				var:uk_rationing_inflation_rate < 15
			}
			change_global_variable = { name = currency_gbp_v divide = 100 }
			change_global_variable = { name = currency_gbp_v multiply = 90 }
			set_variable = { name = uk_rationing_inflation_rate value = 1 }
		}
		else_if = {
			limit = {
				var:uk_rationing_inflation_rate >= 15
				var:uk_rationing_inflation_rate < 20
			}
			change_global_variable = { name = currency_gbp_v divide = 100 }
			change_global_variable = { name = currency_gbp_v multiply = 80 }
			set_variable = { name = uk_rationing_inflation_rate value = 1 }
		}
		else_if = {
			limit = {
				var:uk_rationing_inflation_rate >= 20
				var:uk_rationing_inflation_rate < 25
			}
			change_global_variable = { name = currency_gbp_v divide = 100 }
			change_global_variable = { name = currency_gbp_v multiply = 70 }
			set_variable = { name = uk_rationing_inflation_rate value = 1 }
		}
		else_if = {
			limit = {
				var:uk_rationing_inflation_rate >= 25
				var:uk_rationing_inflation_rate < 30
			}
			change_global_variable = { name = currency_gbp_v divide = 100 }
			change_global_variable = { name = currency_gbp_v multiply = 60 }
			set_variable = { name = uk_rationing_inflation_rate value = 1 }
		}
		else_if = {
			limit = {
				var:uk_rationing_inflation_rate >= 30
			}
			change_global_variable = { name = currency_gbp_v divide = 100 }
			change_global_variable = { name = currency_gbp_v multiply = 50 }
			set_variable = { name = uk_rationing_inflation_rate value = 1 }
		}
		
		ai_chance = {
			base = 100
		}
	}
}

united_kingdom_post_war_events.902 {
	# Determine whether goods need to be rationed or taken off rationing
	type = country_event
	hidden = yes
	
	immediate = {
		if = {
			limit = { has_variable = uk_rationing_aluminum }
			gbr_end_rationing_aluminum = yes
		}
		else_if = { 
			limit = { NOT = { has_variable = uk_rationing_aluminum } }
			gbr_start_rationing_aluminum = yes
		}
		if = { 
			limit = { has_variable = uk_rationing_bauxite }
			gbr_end_rationing_bauxite = yes
		}
		else_if = { 
			limit = { NOT = { has_variable = uk_rationing_bauxite } }
			gbr_start_rationing_bauxite = yes
		}
		if = { 
			limit = { has_variable = uk_rationing_fertilizer }
			gbr_end_rationing_fertilizer = yes
