united_kingdom_post_war_events.101 = {
	# Pound Sterling Devaluation
	# Determine what the new value of the pound sterling will be when devaluing it
	type = country_event
	title = united_kingdom_post_war_events.101.title
	desc = united_kingdom_post_war_events.101.desc
	flavor = united_kingdom_post_war_events.101.flavor
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	event_image = { video = "gfx/event_pictures/middleeast_courtroom_upheaval.bk2" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/middleeast/courtroom_upheaval"
	
	duration = 1
	
	immediate = {
		if = { limit = { inflation_value < 10 }
			change_global_variable = { name = currency_gbp_v multiply = 0.95 } # 5% Devaluation
			change_variable = { name = inflation_rate divide = 10 }
		}
		else_if = {
			limit = {
				inflation_value >= 10
				inflation_value < 20
			}
			change_global_variable = { name = currency_gbp_v multiply = 0.9 } # 10% Devaluation
			change_variable = { name = inflation_rate divide = 10 }
		}
		else_if = {
			limit = {
				inflation_value >= 20
				inflation_value < 30
			}
			change_global_variable = { name = currency_gbp_v multiply = 0.85 } # 15% Devaluation
			change_variable = { name = inflation_rate divide = 10 }
		}
		else_if = {
			limit = {
				inflation_value >= 30
				inflation_value < 40
			}
			change_global_variable = { name = currency_gbp_v multiply = 0.8 } # 20% Devaluation
			change_variable = { name = inflation_rate divide = 10 }
		}
		else_if = {
			limit = {
				inflation_value >= 40
				inflation_value < 50
			}
			change_global_variable = { name = currency_gbp_v multiply = 0.75 } # 25% Devaluation
			change_variable = { name = inflation_rate divide = 10 }
		}
		else_if = {
			limit = { inflation_value >= 50 }
			change_global_variable = { name = currency_gbp_v multiply = 0.7 } # 30% Devaluation
			change_variable = { name = inflation_rate divide = 10 }
		}
		set_variable = { name = gbr_pound_sterling_devalued months = 12 }
	}
	
	option = {
		name = united_kingdom_post_war_events.101.a
		default_option = yes
				
		remove_variable = gbr_devalue_button_pressed
		
		ai_chance = {
			base = 100
		}
	}
}
