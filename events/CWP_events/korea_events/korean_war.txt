### random events triggered by the JE
#Skirmish on the border, initiator
korea_war_events.10 = {
	type = country_event
	placement = scope:korean_border_state
	duration = 3
	title = korea_war_events.10.t
	desc = korea_war_events.10.desc
	flavor = korea_war_events.10.f
	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_election.dds"
	trigger = {
		has_journal_entry = korean_war
		OR = {
			this = c:NRK
			this = c:STK
		}
		NOT = {
			has_global_variable = ongoing_korean_skirmish_event
		}
	}
	cancellation_trigger = {
		NOR = {
			exists = c:NRK
			exists = c:STK
			c:STK = {
				is_at_war = yes
			}
			c:NRK = {
				is_at_war = yes
			}
		}
	}
	immediate = {
		set_global_variable = {
			name = ongoing_korean_skirmish_event
			months = 6
		}
		random_scope_state = {
			limit = {
				owner = c:STK
				any_neighbouring_state = {
					owner = c:NRK
					NOT = {
						owner = root
					}
				}
			}
			save_scope_as = korean_border_state
		}
		random_scope_state = {
			limit = {
				owner = c:NRK
				any_neighbouring_state = {
					owner = c:STK
					NOT = {
						owner = root
					}
				}
			}
			save_scope_as = korean_border_state
		}
	}
	option = {
		# Prolonged skirmish
		name = korea_war_events.10.a
		if = {
			limit = {
				this ?= c:NRK
			}
			c:STK = {
				trigger_event = {
					id = korea_war_events.11
					days = 90
				}
			}
		}
		if = {
			limit = {
				this ?= c:STK
			}
			c:NRK = {
				trigger_event = {
					id = korea_war_events.11
					days = 90
				}
			}
		}
	}
	option = {
		# Attempt to end the skirmish quickly
		name = korea_war_events.10.b
		if = {
			limit = {
				this ?= c:NRK
			}
			c:STK = {
				trigger_event = {
					id = korea_war_events.11
					days = 10
				}
			}
		}
		if = {
			limit = {
				this ?= c:STK
			}
			c:NRK = {
				trigger_event = {
					id = korea_war_events.11
					days = 10
				}
			}
		}
	}
}

		}
