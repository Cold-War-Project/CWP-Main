generic_aerial_bleriotxi_01_diorama_entity = {
	entity = "generic_aerial_bleriotxi_01_entity"

	group = "aerialrecon"

	trigger = {
		scope:military_formation = {
			OR = {
				scope:combat_unit_type = unit_type:combat_unit_type_wartime_aircraft
				scope:combat_unit_type = unit_type:combat_unit_type_fighter_jet
				scope:combat_unit_type = unit_type:combat_unit_type_jet_bombers
				scope:combat_unit_type = unit_type:combat_unit_type_strike_fighter
				scope:combat_unit_type = unit_type:combat_unit_type_attack_aircraft
				scope:combat_unit_type = unit_type:combat_unit_type_fighter_bomber
				scope:combat_unit_type = unit_type:combat_unit_type_stealth_multi_role_aircraft
			}
		}
	}
